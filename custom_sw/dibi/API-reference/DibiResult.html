Class DibiResult

dibi result set.

$result = dibi::query('SELECT * FROM [table]');

$row   = $result->fetch();
$value = $result->fetchSingle();
$table = $result->fetchAll();
$pairs = $result->fetchPairs();
$assoc = $result->fetchAssoc('id');
$assoc = $result->fetchAssoc('active,#,id');

unset($result);
